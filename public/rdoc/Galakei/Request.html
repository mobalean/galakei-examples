def galakei_uid
  if docomo?
    env["HTTP_X_DCMGUID"]
  elsif au?
    env["HTTP_X_UP_SUBNO"]
  elsif softbank?
    env["HTTP_X_JPHONE_UID"]
  end
end

def imode_browser_1_0?
  if /^DoCoMo(.*\((.*;)?c(\d+)\;)?/ =~ user_agent
    $3.to_i < 500
  else
    false
  end
end
